
/*		ad2d1
*		Use RL mod and cn field with indirect descriptors
*		add 9 bit scaled to 9 bit scaled
*		two character offset
*
*/


inst	ad2d	-nt "9scaled+9scal->9scaled"
-io 2  -ir ng
-mf1 ar rl 10 idb reg
-mf2 ar rl 10 idb reg;

desc1 -cn 1  -sd t ;
desc2 -cn 1  -sd t ;

data 1 -do -9 " " (9) "2" "-";
data 2 -do -9 "*" (9) "1" "+";
data 3 "*" (9) "1" "-";

page -all;
